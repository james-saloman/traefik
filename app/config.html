Sample docker-compose.yml

A minimal Traefik + Ignition Gateway setup — plain HTTP routing only, no TLS or certificate resolver. Every line is commented so you can see what each one is actually doing.

# traefik-ignition.docker-compose.yml — a minimal Traefik + Ignition Gateway setup

services:

  # --------------------------------------------------------------
  # Traefik — the reverse proxy / edge router
  # --------------------------------------------------------------
  traefik:
    image: traefik:v3.1                                   # lock to one version — "latest" can quietly break things on upgrade
    container_name: traefik
    command:
      - "--providers.docker=true"                         # let Traefik watch Docker for containers to route to
      - "--providers.docker.exposedbydefault=false"        # only route containers that ask for it with the traefik.enable=true label
      - "--entrypoints.web.address=:80"                    # open port 80 for plain HTTP, call this entrypoint "web"
    ports:
      - "80:80"                                            # make the "web" entrypoint reachable from outside the container
    volumes:
      - "/var/run/docker.sock:/var/run/docker.sock:ro"      # read-only peek at Docker so Traefik can see other containers' labels
    networks:
      - proxy                                               # shared network so Traefik can reach other containers by name
    restart: unless-stopped                                # start back up automatically after a crash or host reboot

  # --------------------------------------------------------------
  # Ignition Gateway — the backend service Traefik routes to
  # --------------------------------------------------------------
  ignition:
    image: inductiveautomation/ignition:latest             # official Ignition Gateway image
    container_name: ignition-gateway
    environment:
      ACCEPT_IGNITION_EULA: "Y"                             # required — the container refuses to start without this
      GATEWAY_ADMIN_USERNAME: admin                          # the admin username Ignition creates on first boot
      GATEWAY_ADMIN_PASSWORD: ${IGNITION_ADMIN_PASSWORD:-changeme}  # pulled from your .env file, or "changeme" if you never set one
    networks:
      - proxy                                               # same network as Traefik, so Traefik can reach it by container name
    labels:
      - "traefik.enable=true"                                            # turn on Traefik routing for this container
      - "traefik.http.routers.ignition.rule=Host(`ignition.internal`)"   # send requests for this hostname to this container
      - "traefik.http.routers.ignition.entrypoints=web"                  # accept these requests on the :80 entrypoint
      - "traefik.http.services.ignition.loadbalancer.server.port=8088"   # the port inside the container that Ignition is actually listening on
    restart: unless-stopped

networks:
  proxy:
    driver: bridge                                          # Docker's standard network type, keeps these containers on their own network
