Sample Configuration

Sample docker-compose.yml

This is the exact shape of the setup shown in the Animations page — Traefik as the edge router, an Ignition Gateway container as the backend. Every line is commented so you can see what each one is actually doing.

# traefik-ignition.docker-compose.yml — minimal Traefik + Ignition Gateway setup

services:

  # --------------------------------------------------------------
  # Traefik — the reverse proxy / edge router
  # --------------------------------------------------------------
  traefik:
    image: traefik:v3.1                                   # pin a version; ":latest" can break configs silently on upgrade
    container_name: traefik
    command:
      - "--providers.docker=true"                         # watch the Docker API for containers to route to
      - "--providers.docker.exposedbydefault=false"        # only route containers that opt in with traefik.enable=true
      - "--entrypoints.web.address=:80"                    # plain HTTP entrypoint, named "web"
      - "--entrypoints.websecure.address=:443"             # HTTPS entrypoint, named "websecure"
      - "--entrypoints.web.http.redirections.entrypoint.to=websecure"       # send all :80 traffic to :443...
      - "--entrypoints.web.http.redirections.entrypoint.scheme=https"       # ...rewriting the scheme to https
      - "--certificatesresolvers.leresolver.acme.httpchallenge=true"        # prove domain ownership via HTTP-01 challenge
      - "--certificatesresolvers.leresolver.acme.httpchallenge.entrypoint=web"   # challenge is served on the :80 entrypoint
      - "--certificatesresolvers.leresolver.acme.email=devops@example.com"       # required contact address for Let's Encrypt
      - "--certificatesresolvers.leresolver.acme.storage=/letsencrypt/acme.json" # where issued certs get cached on disk
    ports:
      - "80:80"                                            # expose entrypoint "web" to the host
      - "443:443"                                           # expose entrypoint "websecure" to the host
    volumes:
      - "/var/run/docker.sock:/var/run/docker.sock:ro"      # read-only access to Docker so Traefik can see container labels
      - "./letsencrypt:/letsencrypt"                        # persist certs across container restarts
    networks:
      - proxy                                               # shared network so Traefik can reach other containers by name
    restart: unless-stopped                                # come back up after a host reboot or crash

  # --------------------------------------------------------------
  # Ignition Gateway — the backend service Traefik routes to
  # --------------------------------------------------------------
  ignition:
    image: inductiveautomation/ignition:latest             # official Ignition Gateway image
    container_name: ignition-gateway
    environment:
      ACCEPT_IGNITION_EULA: "Y"                             # required — image refuses to start without accepting the EULA
      GATEWAY_ADMIN_USERNAME: admin                          # initial gateway admin username
      GATEWAY_ADMIN_PASSWORD: ${IGNITION_ADMIN_PASSWORD:-changeme}  # from .env / shell, falls back to "changeme"
      GATEWAY_RANDOM_ADMIN_PASSWORD: "false"                 # "true" would auto-generate a password on first boot instead
      TZ: Asia/Kolkata                                       # container timezone, affects gateway logs/scheduling
    volumes:
      - ignition-data:/usr/local/bin/ignition/data           # persist projects, tags, and config across container recreation
    networks:
      - proxy                                               # same network as Traefik, so it's reachable by container name
    labels:
      - "traefik.enable=true"                                            # opt this container into Traefik routing
      - "traefik.http.routers.ignition.rule=Host(`ignition.internal`)"   # route requests for this hostname here
      - "traefik.http.routers.ignition.entrypoints=websecure"            # only accept this route on :443
      - "traefik.http.routers.ignition.tls.certresolver=leresolver"      # auto-issue/renew TLS cert via the resolver above
      - "traefik.http.services.ignition.loadbalancer.server.port=8088"   # Ignition's internal HTTP port to forward to
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8088/StatusPing"]  # Ignition's built-in liveness endpoint
      interval: 30s                                          # check every 30s
      timeout: 5s                                            # give each check 5s to respond
      retries: 5                                             # mark unhealthy after 5 consecutive failures
    restart: unless-stopped

networks:
  proxy:
    driver: bridge                                          # Docker's default isolated network driver

volumes:
  ignition-data:                                          # named volume, survives `docker compose down`
